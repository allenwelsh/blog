### 二叉树的中序遍历(非递归)

```js
var inorderTraversal = function (root) {
  if (root === null) {
    return [];
  }
  let result = [];
  let stack = [];
  let p = root;
  while (p) {
    if (p.left !== null) {
      let left = p.left;
      p.left = null;
      stack.push(p);
      p = left; //将删除后的保存到堆栈
    } else if (p.right !== null) {
      result.push(p.val); //没有左子树
      p = p.right;
    } else {
      result.push(p.val);
      p = stack.pop();
    }
  }
  return result;
};
```

### 层次遍历

```js
var levelOrder = function (root) {
  let result = [];
  let queue = [];
  queue.push(root);
  while (queue.length) {
    const len = queue.length;
    let level = [];
    for (let i = 0; i < len; i++) {
      let p = queue.shift();
      level.push(p.val);
      p.left && queue.push(p.left);
      p.right && queue.push(p.right);
    }
    result.push(level);
  }
  return result;
};
```

### 翻转二叉树

```js
var invertTree = function (root) {
  function _invertTree(subRoot) {
    if (subRoot === null) {
      return;
    }
    let tpl = subRoot.left;
    subRoot.left = subRoot.right;
    subRoot.right = tpl;
    _invertTree(subRoot.left);
    _invertTree(subRoot.right);
  }
  _invertTree(root);
  return root;
};
```

#### 二叉搜索树

> 二叉搜索树中的插入操作

```js
var insertIntoBST = function (root, val) {
  let node = new TreeNode(val);
  if (root === null) {
    return node;
  }
  function _insert(subRoot) {
    let left = subRoot.left;
    let right = subRoot.right;
    let sVal = subRoot.val;
    if (sVal >= val) {
      //val大于根节点
      if (left) {
        //判断左节点是否存在
        _insert(left);
      } else {
        //左节点不存在
        subRoot.left = node;
      }
    } else {
      if (right) {
        //判断左节点是否存在
        _insert(right);
      } else {
        //左节点不存在
        subRoot.right = node;
      }
    }
  }
  _insert(root);
  return root;
};
```
